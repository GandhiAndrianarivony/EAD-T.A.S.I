% Title format
% ==============================

  {\MakeUppercase}

  {\normalfont\bfseries\fontsize{12pt}{12pt}\selectfont}
  {\thesection}
  {1em}
  {}

  {\normalfont\bfseries\fontsize{12pt}{12pt}\selectfont}

  {1em}
  {}

% Subsubsection (optional)

  {\normalfont\bfseries\fontsize{12pt}{12pt}\selectfont}
  {\thesubsubsection}

  {}
